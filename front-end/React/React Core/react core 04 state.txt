*** Class components *** used for data/logical and jsx layer ***

import react from 'react';

***** old method ***

class AppName extends React.component {
	constructor(props) {
    super(props);
    this.state = {date: new Date()};
    }
	render(){
		return <div> {this.state.date} </div>
	}
}

*** new method ***

class AppName extends React.component {
	
    state = {date: new Date()};
    }
	render(){
		return <div> {this.state.date} </div>
	}
}
export default AppName;

*** state update ***

class AppName extends React.component {
	state = {count: 0 };
    }
	render(){
		return <button onClick={()=>{this.setState({count: count+1})}}> {this.state.date} </button>
	}
}

*** this.setState()  used for update state.
